.ORG 2000H
DATA SEGMENT
STR1 DB "HELLO"
KEY DB "X"
RES DB 00H
COUNT DB 05
DATA ENDS
CODE SEGMENT
        ASSUME CS:CODE , DS:DATA ,ES:DATA
                MOV AX,DATA
                MOV DS,AX
                MOV ES,AX
                MOV CL,COUNT
                LEA DI ,STR1
                MOV AL,KEY
L2:             SCASB
                JNZ L1
                MOV RES ,01H
                JMP L3
L1:             INC DI
                DEC CL
                JNZ L2
L3:             HLT
CODE ENDS
END
